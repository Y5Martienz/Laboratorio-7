Practica 1

#En el servidor RHEL, instalamos la herramienta NFS:

sudo dnf install nfs-utils -y

#Luego iniciamos, habilitamos y verificamos el estado del servicio:

sudo systemctl start nfs-server
sudo systemctl enable nfs-server
sudo systemctl status nfs-server

#Creamos el directorio en el que ubicaremos los archivos a compartir, y creamos los archivos:

sudo mkdir -p /var/nfs/os3
cd /var/nfs/os3
sudo touch Adrian{1..100}.txt

#Asignamos los permisos apropiados para que el directorio sea accesible por otros usuarios:

sudo chown nobody:nobody /var/nfs/os3
sudo chmod 755 /var/nfs/os3

#Configurarmos el archivo /etc/exports para compartir la carpeta. Agregamos la línea después del ">", ajustando la ip a nuestra dirección de red:

sudo nano /etc/exports
> /var/nfs/os3 192.168.10.17/24(rw,sync,no_subtree_check)

#Guardamos, salimos, y exportamos los recursos compartidos:

sudo exportfs -rav

#En caso de tener el firewalld activo, nos aseguramos de permitir el acceso:

sudo firewall-cmd --permanent --add-service=nfs
sudo firewall-cmd --reload

#Desde el cliente (un clon de RHEL) descargamos la herramienta cliente NFS:

sudo dnf update
sudo dnf install nfs-utils -y

#Creamos un punto de montaje:

sudo mkdir -p /mnt/os3

#Montamos manualmente el recurso NFS (colocamos la ip de nuestro servidor) y verificamos:

sudo mount -t nfs 192.168.10.16:/var/nfs/os3 /mnt/os3
df -h

#Luego, para hacer que el montaje sea permanente, editamos el archivo /etc/fstab y agregamos la línea posterior al signo ">":

sudo nano /etc/fstab
> 192.168.10.16:/var/nfs/os3 /mnt/os3 nfs defaults 0 0

#Guardamos y salimos. Listamos los archivos:

ls /mnt/os3

#Reiniciamos la máquina y, tras el reinicio, verificamos los archivos montados y volvemos a listar los archivos:

sudo reboot
df -h
ls /mnt/os3


Practica 2

#En nuestro servidor RHEL, instalamos SAMBA y herramientas necesarias:

sudo dnf install samba samba-client samba-common -y

#Habilitamos e iniciamos los servicios SMB y NMB:

sudo systemctl enable --now smb nmb

#Creamos la carpeta compartida y asignamos permisos para que todos puedan acceder a los archivos. Luego, creamos los archivos a compartir:

sudo mkdir -p /srv/samba/public
sudo chcon -t samba_share_t /srv/samba/public
cd /srv/samba/public
sudo touch Adrian{1..100}.txt
sudo chmod -R 777 /srv/samba/public
sudo chown -R nobody:nobody /srv/samba/public
ls
ll

#Configuramos el Firewall para SAMBA:

sudo systemctl enable --now smb nmb

sudo firewall-cmd --add-service=samba --permanent
sudo firewall-cmd --reload

#Editamos el archivo de configuración de SAMBA, agregamos el contenido posterior al signo ">", adaptándolo a nuestro archivo compartido:

sudo nano /etc/samba/smb.conf
> despues de [include] borras todo y colocas lo siguiente:
[public]
   path = /srv/samba/public
   browsable = yes
   writable = yes
   guest ok = yes
   read only = no
   force user = nobody

#Luego, creamos los grupos y usuarios para SAMBA, asignando la contraseña:

sudo groupadd sambausers
sudo useradd -M -s /sbin/nologin Marcos
sudo smbpasswd -a Marcos
sudo usermod -aG sambausers Marcos

#Reiniciamos los servicios de SAMBA:

sudo systemctl restart smb nmb

#Luego, vamos al cliente Windows. Para conectar la carpeta compartida como unidad de red, nos dirigimos...

Explorador de archivos> Este equipo> Conectar a unidad de red...

#Escribimos:

\\<192.168.10.16>\public

#Marcamos la casilla "Conectar con otras credenciales" y clicamos en finalizar (Colocaremos las credenciales creadas en el servidor).

#Luego, ingresamos al archivo Adrian99.txt y editamos el texto. Lo guardamos, y nos dirigimos al servidor Rocky para comprobar el cambio:

cd /srv/samba/public
cat Adrian99.txt
